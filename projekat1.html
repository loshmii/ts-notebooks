def calcInfoD(col):
    infoD = 0
    unique = np.unique(col)
    
    for val in unique:
        p = sum(col == val) / len(col)
        infoD -= p * np.log2(p)
    return infoD

IG = []
steps = 100

for i in range(X.shape[1]):
    col = X.iloc[:, i]
    unique = np.unique(col)

    if (len(unique) > steps): # group values

def calcR(corr):
    k = corr.shape[1]-1
    rzi = corr.iloc[:-1, -1].mean()
    rij = corr.iloc[:-1, :-1].values.mean() - 1/k
    r = k*rzi/np.sqrt(k+k*(k-1)*rij)
    
    return np.abs(r)

def parametres(X, y, classi):
    Xi = X.loc[y == classi, :]
    priorX = Xi.shape[0] / X.shape[0]
    meanX = Xi.mean().values.reshape(Xi.shape[1], 1)
    covX = Xi.cov()
    return (priorX, meanX, covX)

def trainingTestSplit(x, y, trainSize):
    n = x.shape[0]
    nTrain = int(n * trainSize)
    nTest = n - nTrain

    xTrain = x.iloc[:nTrain].reset_index(drop=True)
    yTrain = y.iloc[:nTrain].reset_index(drop=True)
    xTest = x.iloc[nTrain:].reset_index(drop=True)
    yTest = y.iloc[nTrain:].reset_index(drop=True)

    return xTrain, yTrain, xTest, yTest

def calcPDF(x, mean, var):
    coeff = 1 / np.sqrt(2 * np.pi * var)
    exponent = np.exp(- (x - mean) ** 2 / (2 * var))
    return coeff * exponent
